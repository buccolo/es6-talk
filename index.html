Iterator / Lazy Sequences


let fibonacci = {
    [Symbol.iterator]() {
        let pre = 0, cur = 1

        return {
           next () {
               [ pre, cur ] = [ cur, pre + cur ]

               return { done: false, value: cur }
           }
        }
    }
}

for (let n of fibonacci) {
    if (n > 10) {
      console.log(n)
      break
    }
}

Class

    constructor (id, x, y) {
        this.id = id
        this.move(x, y)
    }

        this.x = x
        this.y = y
    }
}

const square = new Shape(1, 4, 5)
